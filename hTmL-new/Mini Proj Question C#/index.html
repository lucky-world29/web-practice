Answer 1: Employee Salary System

using System;

class Person {
    public string Name;
    public int Age;
    public Person(string name, int age) {
        Name = name; Age = age;
    }
    public override string ToString() {
        return $"Name: {Name}\nAge: {Age}";
    }
}

class Employee : Person {
    public int EmployeeID;
    public string Department;
    public Employee(string name, int age, int empId, string dept) 
        : base(name, age) {
        EmployeeID = empId; Department = dept;
    }
    public override string ToString() {
        return base.ToString() +
               $"\nEmployeeID: {EmployeeID}\nDepartment: {Department}";
    }
}

class Salary : Employee {
    public double BasicSalary, Bonus;
    public Salary(string name, int age, int empId, string dept, double basic, double bonus) 
        : base(name, age, empId, dept) {
        BasicSalary = basic; Bonus = bonus;
    }
    public override string ToString() {
        return base.ToString() +
               $"\nBasic Salary: {BasicSalary}\nBonus: {Bonus}";
    }
}

class Program {
    static void Main() {
        Salary s = new Salary("Vaibhav", 28, 101, "IT", 50000, 7000);
        Console.WriteLine(s);
    }
}

Answer 4: Employee → Manager → Director

using System;

class Employee {
    public int EmpID;
    public string Name;
    public double Salary;
    public Employee(int id, string name, double salary) {
        EmpID = id; Name = name; Salary = salary;
    }
    public override string ToString() =>
        $"EmpID: {EmpID}\nName: {Name}\nSalary: {Salary}";
}

class Manager : Employee {
    public string Department;
    public Manager(int id, string name, double salary, string dept)
        : base(id, name, salary) { Department = dept; }
    public override string ToString() => base.ToString() +
        $"\nDepartment: {Department}";
}

class Director : Manager {
    public double Bonus;
    public Director(int id, string name, double salary, string dept, double bonus) 
        : base(id, name, salary, dept) { Bonus = bonus; }
    public override string ToString() => base.ToString() +
        $"\nBonus: {Bonus}";
}

class Program {
    static void Main() {
        Console.Write("Enter number of Directors: ");
        int n = int.Parse(Console.ReadLine());
        Director[] arr = new Director[n];
        for (int i = 0; i < n; i++) {
            Console.WriteLine($"Enter details for Director {i+1}:");
            Console.Write("ID: "); int id = int.Parse(Console.ReadLine());
            Console.Write("Name: "); string name = Console.ReadLine();
            Console.Write("Salary: "); double sal = double.Parse(Console.ReadLine());
            Console.Write("Department: "); string dept = Console.ReadLine();
            Console.Write("Bonus: "); double bonus = double.Parse(Console.ReadLine());
            arr[i] = new Director(id, name, sal, dept, bonus);
        }
        Console.WriteLine("\n--- Directors List ---");
        foreach (var d in arr) Console.WriteLine(d + "\n");
    }
}

Answer 5: University Student Result System

using System;

class Person {
    public string Name; public int Age;
    public Person(string name, int age) { Name = name; Age = age; }
    public override string ToString() => $"Name: {Name}\nAge: {Age}";
}

class Student : Person {
    public int RollNo; public string Course;
    public Student(string name, int age, int roll, string course) : base(name, age) {
        RollNo = roll; Course = course;
    }
    public override string ToString() => base.ToString() +
        $"\nRollNo: {RollNo}\nCourse: {Course}";

class Result : Student {
    public int M1, M2, M3;
    public Result(string name, int age, int roll, string course, int m1, int m2, int m3) 
        : base(name, age, roll, course) { M1 = m1; M2 = m2; M3 = m3; }

        return base.ToString() + $"\nMarks: {M1}, {M2}, {M3}\nTotal: {total}\n%: {percent:F2}";
    }
}

Answer 6: Real Estate Property Sale System

using System;

class Property {
    public int PropertyID; public string Location;
    public double BasePrice;
    public Property(int id, string loc, double price) {
        PropertyID = id; Location = loc; BasePrice = price;
    }
    public override string ToString() => $"PropertyID: {PropertyID}\nLocation: {Location}\nBasePrice: {BasePrice}";
}

class House : Property {
    public string Type; public double Area;
    public House(int id, string loc, double price, string type, double area) 
        : base(id, loc, price) { Type = type; Area = area; }
    public override string ToString() => base.ToString() +
        $"\nType: {Type}\nArea: {Area} sq.ft";
}

class Sale : House {
    public double TaxPercent, DiscountPercent;
    public Sale(int id, string loc, double price, string type, double area, double tax, double disc) 
        : base(id, loc, price, type, area) { TaxPercent = tax; DiscountPercent = disc; }
    public double FinalPrice() {
        double taxAmt = BasePrice * TaxPercent / 100;
        double discAmt = BasePrice * DiscountPercent / 100;
        return BasePrice + taxAmt - discAmt;
    }
    public override string ToString() => base.ToString() +
        $"\nTax: {TaxPercent}%\nDiscount: {DiscountPercent}%\nFinal Price: {FinalPrice()}";
}

class Program {
    static void Main() {
        Sale s = new Sale(1, "Pune", 5000000, "Villa", 2500, 5, 2);
        Console.WriteLine(s);
    }
}

Answer 7: Product Order Invoice System

using System;

class Product {
    public int ProductID; public string Name; public double Price;
    public Product(int id, string name, double price) { ProductID = id; Name = name; Price = price; }
    public override string ToString() => $"ProductID: {ProductID}\nName: {Name}\nPrice: {Price}";
}

class Order : Product {
    public int Quantity; public string CustomerName;
    public Order(int id, string name, double price, int qty, string cust) 
        : base(id, name, price) { Quantity = qty; CustomerName = cust; }
    public override string ToString() => base.ToString() +
        $"\nQuantity: {Quantity}\nCustomer: {CustomerName}";
}

class Invoice : Order {
    public double TaxPercent, DiscountPercent;
    public Invoice(int id, string name, double price, int qty, string cust, double tax, double disc) 
        : base(id, name, price, qty, cust) { TaxPercent = tax; DiscountPercent = disc; }
    public double FinalBill() {
        double subtotal = Price * Quantity;
        double taxAmt = subtotal * TaxPercent / 100;
        double discAmt = subtotal * DiscountPercent / 100;
        return subtotal + taxAmt - discAmt;
    }
    public override string ToString() => base.ToString() +
        $"\nTax: {TaxPercent}%\nDiscount: {DiscountPercent}%\nFinal Bill: {FinalBill()}";
}

class Program {
    static void Main() {
        Invoice inv = new Invoice(1, "Laptop", 60000, 2, "Neha", 18, 5);
        Console.WriteLine(inv);
    }
}
